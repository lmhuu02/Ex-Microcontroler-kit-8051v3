#include <REGX52.H>
#include <string.h>
//ham delay
void delay(int time){
while(time--){}
}
//ham gui mot ky tu
void send(unsigned char a){
	SBUF = a;
	while(!TI){}	//kiem tra nhan
		TI = 0;
}
//ham gui chuoi ky tu
void sendChar(unsigned char *a){
	int i,x;
	x = strlen(a);
	for(i = 0; i <= x; i++){
			send(a[i]);
	}
}
//ham nhan ky tu tu may tinh

	b = SBUF;
	while(!RI){}	//kiem tra truyen
		RI = 0;
}
